229     /**
230      * {@inheritDoc}
231      */
232     protected void updateExtensionFields()
233         throws MojoExecutionException
234     {
235         populateMissingFields();
236 
237         if( null == m_candidateIds )
238         {
239             setMainArchetype( OSGI_WRAPPER_ARCHETYPE_ID );
240 
241             // only need to set this once: the same groupId is used for extra wrappers
242             getArchetypeMojo().setField( "groupId", getInternalGroupId( bundleGroupId ) );
243 
244             // bootstrap with the initial wrapper artifact
245             String rootId = groupId + ':' + artifactId + ':' + version;
246 
247             m_candidateIds = new ArrayList();
248             m_excludedIds = new HashSet();
249             m_wrappedIds = new HashSet();
250 
251             excludeCandidates( exclusions );
252 
253             // kickstart the wrapping
254             m_candidateIds.add( rootId );
255             m_wrappedIds.add( rootId );
256         }
257 
258         String id = (String) m_candidateIds.remove( 0 );
259         String[] fields = id.split( ":" );
260 
261         groupId = fields[0];
262         artifactId = fields[1];
263         version = fields[2];
264 
265         String compoundWrapperId = getBundleSymbolicName();
266         if( addVersion )
267         {
268             compoundWrapperId += '-' + version;
269         }
270 
271         // common groupId shared between all wrappers in same session
272         getArchetypeMojo().setField( "artifactId", compoundWrapperId );
273         getArchetypeMojo().setField( "packageName", artifactId );
274         getArchetypeMojo().setField( "version", version );
275     }

432     /**
433      * Attempt to wrap direct dependencies of the wrapped artifact - in turn they will call this method, and so on...
434      * 
435      * @param pom Maven project model
436      */
437     private void wrapDirectDependencies( Pom pom )
438     {
439         /*
440          * Use a local list to capture dependencies that are type POM, ie. collections of dependencies. These POM
441          * artifacts don't require wrapping, so we must store and process them locally in the following loop...
442          */
443         List dependencyPoms = new ArrayList();
444 
445         // use the wrapped artifact's POM to kick things off
446         dependencyPoms.add( getFactory().createProjectArtifact( groupId, artifactId, version ) );
447 
448         while( !dependencyPoms.isEmpty() )
449         {
450             Artifact pomArtifact = (Artifact) dependencyPoms.remove( 0 );
451 
452             try
453             {
454                 // Standard Maven code to get direct dependencies for a given POM
455                 MavenProject p = m_projectBuilder.buildFromRepository( pomArtifact, m_remoteRepos, getLocalRepo() );
456                 Set artifacts = p.createArtifacts( getFactory(), null, null );
457 
458                 // look for new artifacts to wrap
459                 dependencyPoms.addAll( processDependencies( pom, artifacts ) );
460             }
461             catch( ProjectBuildingException e )
462             {
463                 getLog().warn( e );
464             }
465             catch( InvalidDependencyVersionException e )
466             {
467                 getLog().warn( e );
468             }
469         }
470     }

472     /**
473      * Look for more artifacts that need to be wrapped, ignoring those already wrapped or containing OSGi metadata
474      * 
475      * @param pom Maven project model
476      * @param artifacts list of potential artifacts to be wrapped
477      * @return list of POM artifacts discovered while processing
478      */
479     private List processDependencies( Pom pom, Set artifacts )
480     {
481         List newDependencyPoms = new ArrayList();
482         for( Iterator i = artifacts.iterator(); i.hasNext(); )
483         {
484             Artifact artifact = (Artifact) i.next();
485             String candidateId = getCandidateId( artifact );
486 
487             if( isValidWrapperDependency( artifact ) )
488             {
489                 // process POM artifacts in parent loop...
490                 if( "pom".equals( artifact.getType() ) )
491                 {
492                     newDependencyPoms.add( artifact );
493                 }
494                 // copy dependency to current wrapper pom (not all require wrapping)
495                 else if( addWrapperDependency( pom, artifact ) )
496                 {
497                     m_candidateIds.add( candidateId );
498                     m_wrappedIds.add( candidateId );
499                 }
500             }
501         }
502 
503         return newDependencyPoms;
504     }

506     /**
507      * Get the first version of the given artifact to be wrapped during this cycle
508      * 
509      * @param candidateId potential new candidate
510      * @return first version wrapped, or null if it has not been wrapped
511      */
512     private String getWrappedVersion( String candidateId )
513     {
514         // ignore version field while searching...
515         int versionIndex = candidateId.lastIndexOf( ':' );
516         String prefix = candidateId.substring( 0, 1 + versionIndex );
517 
518         for( Iterator i = m_wrappedIds.iterator(); i.hasNext(); )
519         {
520             String wrappedId = (String) i.next();
521             if( wrappedId.startsWith( prefix ) )
522             {
523                 // return version field from the id
524                 return wrappedId.substring( prefix.length() );
525             }
526         }
527 
528         return null;
529     }

540     /**
541      * @param artifact wrapper dependency
542      * @return true if this artifact needs to be wrapped, otherwise false
543      */
544     private boolean isValidWrapperDependency( Artifact artifact )
545     {
546         String scope = artifact.getScope();
547 
548         if( Artifact.SCOPE_SYSTEM.equals( scope ) || Artifact.SCOPE_TEST.equals( scope ) )
549         {
550             getLog().info( "Skipping dependency " + artifact );
551             return false;
552         }
553         else if( !wrapOptional && artifact.isOptional() )
554         {
555             getLog().info( "Skipping optional dependency " + artifact );
556             return false;
557         }
558 
559         return true;
560     }

562     /**
563      * Adds a dependency to the wrapped artifact, or the original if it doesn't require wrapping
564      * 
565      * @param pom Maven project model
566      * @param artifact wrapper dependency
567      * @return true if the dependency should be wrapped, otherwise false
568      */
569     private boolean addWrapperDependency( Pom pom, Artifact artifact )
570     {
571         if( m_excludedIds.contains( artifact.getGroupId() + ':' + artifact.getArtifactId() ) )
572         {
573             // exclude this dependency from current POM rather than wrapping it here
574             pom.addExclusion( artifact.getGroupId(), artifact.getArtifactId(), true );
575             return false;
576         }
577         else if( PomUtils.isBundleArtifact( artifact, getResolver(), m_remoteRepos, getLocalRepo(), testMetadata ) )
578         {
579             pom.addDependency( getBundleDependency( artifact ), true );
580             return false;
581         }
582         else
583         {
584             String existingVersion = getWrappedVersion( getCandidateId( artifact ) );
585             if( null != existingVersion )
586             {
587                 /*
588                  * We've already wrapped this artifact but with a different version: we can't easily update any of the
589                  * earlier wrapper POMs so for now we'll force the other transitive wrappers to use the first version
590                  */
591                 artifact.setVersion( existingVersion );
592             }
593 
594             pom.addDependency( getWrappedDependency( artifact ), true );
595             return ( null == existingVersion );
596         }
597     }

615     /**
616      * @param artifact wrapper dependency
617      * @return a provided dependency to the (future) wrapped artifact
618      */
619     private Dependency getWrappedDependency( Artifact artifact )
620     {
621         Dependency dependency = new Dependency();
622 
623         dependency.setGroupId( getInternalGroupId( bundleGroupId ) );
624         String compoundWrapperId = getCompoundId( artifact.getGroupId(), artifact.getArtifactId() );
625         String metaVersion = PomUtils.getMetaVersion( artifact );
626 
627         if( addVersion )
628         {
629             dependency.setArtifactId( compoundWrapperId + '-' + metaVersion );
630             dependency.setVersion( INITIAL_POM_VERSION );
631         }
632         else
633         {
634             dependency.setArtifactId( compoundWrapperId );
635             dependency.setVersion( metaVersion + '-' + INITIAL_POM_VERSION );
636         }
637 
638         dependency.setScope( Artifact.SCOPE_PROVIDED );
639 
640         return dependency;
641     }

643     /**
644      * Explicitly exclude artifacts from the wrapping process
645      * 
646      * @param artifacts comma-separated list of artifacts to exclude from wrapping
647      */
648     private void excludeCandidates( String artifacts )
649     {
650         if( null == artifacts || artifacts.length() == 0 )
651         {
652             return;
653         }
654 
655         String[] exclusionIds = artifacts.split( "," );
656         for( int i = 0; i < exclusionIds.length; i++ )
657         {
658             String id = exclusionIds[i].trim();
659             String[] fields = id.split( ":" );
660             if( fields.length > 1 )
661             {
662                 // handle groupId:artifactId:other:stuff
663                 m_excludedIds.add( fields[0] + ':' + fields[1] );
664             }
665             else
666             {
667                 // assume groupId same as artifactId
668                 m_excludedIds.add( id + ':' + id );
669             }
670         }
671     }

673     /**
674      * Add explicit dependency exclusion list to the main POM dependency element
675      * 
676      * @param pom Maven project model
677      */
678     private void excludeDependencies( Pom pom )
679     {
680         for( Iterator i = m_excludedIds.iterator(); i.hasNext(); )
681         {
682             String[] fields = ( (String) i.next() ).split( ":" );
683             pom.addExclusion( fields[0], fields[1], true );
684         }
