43  /**
44   * Updates a Pax-Construct project or (when run in the scripts directory) the installed scripts to the latest version
45   * 
46   * <code><pre>
47   *   mvn pax:update [-Dversion=...]
48   * </pre></code>
49   * 
50   * @goal update
51   * @aggregator true
52   * 
53   * @requiresProject false
54   */
55  public class UpdateMojo extends AbstractMojo
56  {
57      /**
58       * Component factory for Maven artifacts
59       * 
60       * @component
61       */
62      private ArtifactFactory m_factory;

65       * Component for resolving Maven artifacts
66       * 
67       * @component
68       */
69      private ArtifactResolver m_resolver;

73       * 
74       * @component
75       */
76      private ArtifactMetadataSource m_source;

78      /**
79       * Component factory for Maven repositories.
80       * 
81       * @component

83      private ArtifactRepositoryFactory m_repoFactory;

91       * List of remote Maven repositories for the containing project.
92       * 
93       * @parameter expression="${project.remoteArtifactRepositories}"
94       * @required
95       * @readonly
96       */
97      private List m_remoteRepos;

99      /**
100      * The local Maven repository for the containing project.
101      * 
102      * @parameter expression="${localRepository}"
103      * @required
104      * @readonly
105      */
106     private ArtifactRepository m_localRepo;

108     /**
109      * The directory containing the POM to be updated.
110      * 
111      * @parameter expression="${targetDirectory}" default-value="${project.basedir}"
112      */
113     private File targetDirectory;
114 
115     /**
116      * The version of Pax-Construct to update to.
117      * 
118      * @parameter expression="${version}"
119      */
120     private String version;
121 
122     /**
123      * {@inheritDoc}
124      */
125     public void execute()
126         throws MojoExecutionException
127     {
128         String groupId = AbstractPaxArchetypeMojo.PAX_CONSTRUCT_GROUP_ID;
129 
130         // find latest release if no explicit version is given
131         Artifact scripts = m_factory.createBuildArtifact( groupId, "scripts", version, "zip" );
132         if( PomUtils.needReleaseVersion( version ) )
133         {
134             version = PomUtils.getReleaseVersion( scripts, m_source, m_remoteRepos, m_localRepo, null );
135             scripts.selectVersion( version );
136         }
137 
138         if( new File( targetDirectory, "pax-bootstrap-pom.xml" ).exists() )
139         {
140             updatePaxConstructScripts( scripts );
141         }
142         else if( new File( targetDirectory, "pom.xml" ).exists() )
143         {
144             updatePaxConstructProject();
145         }
146         else
147         {
148             getLog().warn( "pax-update should be run from the scripts directory, or from a Pax-Construct project" );
149         }
150     }
151 
152     /**
153      * Update each script in turn from the zipfile stored in the repository
154      * 
155      * @param scripts maven artifact pointing to the scripts
156      * @throws MojoExecutionException
157      */
158     private void updatePaxConstructScripts( Artifact scripts )
159         throws MojoExecutionException
160     {
161         // download to local repository if not already there
162         if( !PomUtils.downloadFile( scripts, m_resolver, m_remoteRepos, m_localRepo ) )
163         {
164             throw new MojoExecutionException( "Unable to download scripts " + scripts );
165         }
166 
167         getLog().info( "Updating scripts to version " + version );
168 
169         try
170         {
171             ZipFile zip = new ZipFile( scripts.getFile() );
172             for( Enumeration i = zip.entries(); i.hasMoreElements(); )
173             {
174                 ZipEntry entry = (ZipEntry) i.nextElement();
175 
176                 // only need shallow copying (no nested folders)
177                 String name = new File( entry.getName() ).getName();
178                 String path = new File( targetDirectory, name ).getPath();
179 
180                 if( !entry.isDirectory() )
181                 {
182                     // overwrite using data stored in the zip
183                     InputStream in = zip.getInputStream( entry );
184                     String data = IOUtil.toString( in );
185                     FileUtils.fileWrite( path, data );
186 
187                     getLog().info( " => " + path );
188                 }
189             }
190         }
191         catch( IOException e )
192         {
193             throw new MojoExecutionException( "Problem updating local scripts", e );
194         }
195     }
196 
197     /**
198      * Update existing Pax-Construct project to use the latest release of the plugin
199      * 
200      * @throws MojoExecutionException
201      */
202     private void updatePaxConstructProject()
203         throws MojoExecutionException
204     {
205         try
206         {
207             // only need to update in one place...
208             Pom pom = PomUtils.readPom( targetDirectory );
209             boolean foundPlugin = pom.updatePluginVersion( "org.ops4j", "maven-pax-plugin", version );
210             if( foundPlugin )
211             {
212                 getLog().info( "Updating Pax-Construct project to version " + version );
213 
214                 pom.write();
215 
216                 getLog().info( " => " + pom.getFile() );
217             }
218             else
219             {
220                 getLog().warn( "Unable to find reference to org.ops4j:maven-pax-plugin" );
221                 getLog().warn( "you may need to convert this project using 'pax-clone'" );
222             }
223         }
224         catch( IOException e )
225         {
226             throw new MojoExecutionException( "Problem reading Maven POM: " + targetDirectory );
227         }
228     }
